Sites Service : Email Template using a Webpage


Sites Service: Email Template using a Webpage:


A Google Apps Script standalone 'Email Template' using a Google Sites webpage.

Using the 'doGet' so the Apps script can be used as a Apps Script Gadget directly from the Google Sites Insert Menu. 

  • Add the Apps script Snippet to Google Site.
  • Add Email /replyTo Email [Set to Active User email]
  • Save/Authorize the App Script.
  • Insert the Apps script gadget from the Insert menu in google sites.

Contact form UI 
  • Add Contact Details 
Submit Email using Google sites Template, replacing the form data.

You can the set the page level permissions to hide the website template from the public, but still access it through Google Apps Script.



Submit an Email Address to receive a Google Sites 'Templated' Email

Add Apps Script to your Google sites


Add the Apps Script Snippet [below] to your Google Sites - Apps Script Editor . 

Insert into your Google sites page and set the Apps Script Gadget properties to allow anonymous users :

Apps Script Gadget properties > Allow anyone to invoke this service and [tick] Allow anonymous access

[You Must provide a valid email to preview the results]

Simple Email Template using webpage
SiteTemplate for Email


Apps Script Snippet:


var app = UiApp.createApplication().setTitle('Email Template using a Google Sites Webpage');

function doGet() {
   
  
  var main_grid = app.createGrid(6, 1).setId('main_grid').setStyleAttribute('color', '#ccc').setStyleAttribute('padding' , '40px 50px');
  main_grid.setStyleAttribute('float', 'center').setWidth('600');
  
  var form_submit_handler = app.createServerClickHandler('form_submit');
  var form_submit = app.createButton('Send Email',form_submit_handler).setId('form_submit');
  form_submit_handler.addCallbackElement(main_grid);
  
  var name_text  = app.createTextBox().setId('name_text').setName('name_text').setValue('Name').setWidth('200');
      
  var email_text = app.createTextBox().setId('email_text').setName('email_text').setValue('Email').setWidth('200');
      
  var phone_text = app.createTextBox().setId('phone_text').setName('phone_text').setValue('Phone').setWidth('200');
    
  var thanks_text = app.createRichTextArea().setId('thanks_text').setText('Thank you! You will recieve a Confirmation Email shortly.').setWidth('350').setHeight('400').setStyleAttribute('color', '#ff6600').setHeight('60');
  thanks_text.setVisible(false);
  
  var email_template = app.createTextArea().setId('email_template').setName('email_template').setValue('Email Template').setWidth('350').setHeight('400');
  email_template.setVisible(false);
  
  //ADD TEMPLATE LOCATION URL
  var customer_email_template = SitesApp.getPageByUrl('https://sites.google.com/a/TEMPLATE_LOCATION').getHtmlContent();
  email_template.setText(customer_email_template);
  
  app
    .add(main_grid)
    .add(thanks_text)
    
  
  main_grid.setWidget(0,0,name_text);
  main_grid.setWidget(1,0,email_text);
  main_grid.setWidget(2,0,phone_text);
  main_grid.setWidget(3,0,form_submit);
  main_grid.setWidget(4,0,email_template);
  return app;
  
}

function form_submit(e) {
  
  var form_name = e.parameter.name_text;
  var form_email = e.parameter.email_text;
  var form_phone = e.parameter.phone_text;
  var form_email_template = e.parameter['email_template'];
    
  
  //Gets the Active User
  var customer_email = Session.getUser().getEmail();
    
  var body = 'Please use a Html Capable email client.'; //ignored when using html in email is available
  
  //ADD FORM DATE TO SITES TEMPLATE for email
  
  var customer_body_html = form_email_template;
  customer_body_html = customer_body_html.replace('${form_name}', form_name);
  customer_body_html = customer_body_html.replace('${form_email}', form_email);
  customer_body_html = customer_body_html.replace('${form_phone}', form_phone);
   
  
// SWAP customer_email [below] for testing and form_email for live submissions.
  
MailApp.sendEmail(customer_email, 'Customer Booking Confirmation', body, {htmlBody:customer_body_html, replyTo:'MY@EMAIL.com'});
   
  app.getElementById('main_grid').setVisible(false);
  
  //View Html Rendered ????
  app.getElementById('thanks_text').setVisible(true);
                           
    
  return app;
}
